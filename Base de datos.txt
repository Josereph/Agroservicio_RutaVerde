-- ============================================
-- BASE DE DATOS: RutaVerdeBD
-- SISTEMA DE GESTIÓN DE AGROSERVICIOS
-- Integración de todos los módulos
-- ============================================

-- Crear y usar la base de datos
CREATE DATABASE RutaVerdeBD;
USE RutaVerdeBD;

-- ============================================
-- MÓDULO: UBICACIONES GEOGRÁFICAS
-- Departamentos, Municipios y Direcciones de El Salvador
-- ============================================municipio

-- Tabla: Departamento
CREATE TABLE Departamento (
    Id_Departamento INT PRIMARY KEY,
    Nombre_Departamento VARCHAR(100) NOT NULL
) ENGINE=InnoDB COMMENT='Departamentos de El Salvador';

-- Tabla: Municipio
CREATE TABLE Municipio (
    Id_Municipio INT PRIMARY KEY,
    Id_Departamento INT NOT NULL,
    Nombre_Municipio VARCHAR(100) NOT NULL,
    FOREIGN KEY (Id_Departamento) REFERENCES Departamento(Id_Departamento)
) ENGINE=InnoDB COMMENT='Municipios de El Salvador';

-- Tabla: Direccion
CREATE TABLE Direccion (
    Id_Direccion INT PRIMARY KEY AUTO_INCREMENT,
    Id_Municipio INT NOT NULL,
    Detalle_Direccion TEXT NULL,
    FOREIGN KEY (Id_Municipio) REFERENCES Municipio(Id_Municipio)
) ENGINE=InnoDB COMMENT='Direcciones específicas';

-- Tabla: Ubicaciones (relaciona departamento, municipio y dirección)
CREATE TABLE Ubicaciones (
    Id_Ubicacion INT PRIMARY KEY AUTO_INCREMENT,
    Id_Departamento INT NOT NULL,
    Id_Municipio INT NOT NULL,
    Id_Direccion INT NOT NULL,
    FOREIGN KEY (Id_Departamento) REFERENCES Departamento(Id_Departamento),
    FOREIGN KEY (Id_Municipio) REFERENCES Municipio(Id_Municipio),
    FOREIGN KEY (Id_Direccion) REFERENCES Direccion(Id_Direccion)
) ENGINE=InnoDB COMMENT='Ubicaciones completas para envíos';

-- ============================================
-- MÓDULO: VEHÍCULOS DE CARGA
-- BY Joseph
-- ============================================

-- Catálogo: Tipo de Vehículo
CREATE TABLE Cat_Tipo_Vehiculo (
    tipo_id TINYINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(30) NOT NULL UNIQUE,
    descripcion VARCHAR(120) NULL
) ENGINE=InnoDB COMMENT='Catálogo de tipos de vehículo';

-- Catálogo: Estado del Vehículo
CREATE TABLE Cat_Estado_Vehiculo (
    estado_id TINYINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(30) NOT NULL UNIQUE,
    es_operativo BOOLEAN NOT NULL DEFAULT TRUE
) ENGINE=InnoDB COMMENT='Catálogo de estados operativos de los vehículos';

-- Tabla principal: Vehiculos
CREATE TABLE Vehiculos (
    id_vehiculo INT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    unidad_numero VARCHAR(10) NOT NULL,
    placa VARCHAR(10) NOT NULL,
    vin VARCHAR(17) NULL,
    tipo_id TINYINT UNSIGNED NOT NULL,
    marca VARCHAR(40) NULL,
    modelo VARCHAR(40) NULL,
    anio SMALLINT NULL,
    capacidad_kg DECIMAL(10,2) NOT NULL,
    estado_id TINYINT UNSIGNED NOT NULL,
    km_actual INT UNSIGNED NOT NULL DEFAULT 0,
    seguro_vigente BOOLEAN NOT NULL DEFAULT FALSE,
    aseguradora VARCHAR(60) NULL,
    poliza_numero VARCHAR(40) NULL,
    fecha_venc_seguro DATE NULL,
    observaciones TEXT NULL,
    fecha_registro TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    CONSTRAINT uq_vehiculo_unidad UNIQUE (unidad_numero),
    CONSTRAINT uq_vehiculo_placa UNIQUE (placa),
    CONSTRAINT uq_vehiculo_vin UNIQUE (vin),
    
    CONSTRAINT fk_vehiculo_tipo FOREIGN KEY (tipo_id) 
        REFERENCES Cat_Tipo_Vehiculo(tipo_id),
    CONSTRAINT fk_vehiculo_estado FOREIGN KEY (estado_id) 
        REFERENCES Cat_Estado_Vehiculo(estado_id),
    
    CONSTRAINT chk_capacidad_pos CHECK (capacidad_kg > 0),
    CONSTRAINT chk_km_no_neg CHECK (km_actual >= 0)
) ENGINE=InnoDB COMMENT='Registro maestro de vehículos de carga del agroservicio';

-- Índices para optimización
CREATE INDEX ix_vehiculo_estado ON Vehiculos(estado_id);
CREATE INDEX ix_vehiculo_tipo ON Vehiculos(tipo_id);
CREATE INDEX ix_vehiculo_capacidad ON Vehiculos(capacidad_kg);

-- ============================================
-- MÓDULO: CONDUCTORES
-- Información de los choferes
-- ============================================

CREATE TABLE conductor (
    id_conductor INT AUTO_INCREMENT PRIMARY KEY,
    nombre_completo VARCHAR(100) NOT NULL,
    documento_identificacion VARCHAR(20) NOT NULL UNIQUE,
    tipo_licencia ENUM('Liviana', 'Pesada', 'Pesada T') NOT NULL,
    fecha_vencimiento_licencia DATE NOT NULL,
    telefono VARCHAR(15) NULL,
    correo VARCHAR(100) NULL,
    estado ENUM('Activo', 'De Vacaciones', 'Suspendido') DEFAULT 'Activo',
    experiencia_notas VARCHAR(255) NULL,
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB COMMENT='Registro de conductores del agroservicio';

-- ============================================
-- MÓDULO: CLIENTES
-- Información de personas o empresas que solicitan servicios
-- ============================================

CREATE TABLE Clientes (
    Id_Cliente INT NOT NULL AUTO_INCREMENT,
    Nombre_Cliente VARCHAR(100) NOT NULL,
    Dui VARCHAR(20) NOT NULL UNIQUE,
    CorreoElectronico VARCHAR(200) NOT NULL,
    PRIMARY KEY (Id_Cliente)
) ENGINE=InnoDB COMMENT='Clientes del agroservicio';

-- ============================================
-- MÓDULO: CATÁLOGOS DE SERVICIOS
-- ============================================

-- Tabla: Tipo de Servicio (Entrega o Retiro)
CREATE TABLE Tipo_Servicio (
    Id_Tipo_Servicio INT NOT NULL AUTO_INCREMENT,
    Nombre_Servicio VARCHAR(20) NOT NULL UNIQUE,
    PRIMARY KEY (Id_Tipo_Servicio)
) ENGINE=InnoDB COMMENT='Tipos de servicio disponibles';

-- Tabla: Nivel de Fragilidad
CREATE TABLE Nivel_Fragilidad (
    Id_Fragilidad INT NOT NULL AUTO_INCREMENT,
    Nivel VARCHAR(30) NOT NULL UNIQUE,
    Detalle_Impacto TEXT,
    PRIMARY KEY (Id_Fragilidad)
) ENGINE=InnoDB COMMENT='Niveles de fragilidad para cargamentos';

-- ============================================
-- MÓDULO: SERVICIOS (TABLA CENTRAL)
-- Registra cada solicitud de envío/transporte
-- ============================================

CREATE TABLE Servicios (
    Id_Servicio INT NOT NULL AUTO_INCREMENT,
    Id_Cliente INT NOT NULL,
    Id_Vehiculo INT UNSIGNED NOT NULL,
    id_conductor INT NOT NULL,
    Id_Tipo_Servicio INT NOT NULL,
    Id_Fragilidad INT NOT NULL,
    Id_Ubicacion INT NOT NULL,
    Peso_Carga DECIMAL(10, 2) NOT NULL,
    Fecha_Pedido DATE NOT NULL,
    Fecha_Entrega DATE NOT NULL,
    Precio_Total DECIMAL(10, 2) NOT NULL,
    
    PRIMARY KEY (Id_Servicio),
    
    CHECK (Peso_Carga > 0),
    CHECK (Precio_Total >= 0),
    
    FOREIGN KEY (Id_Cliente) REFERENCES Clientes(Id_Cliente),
    FOREIGN KEY (Id_Vehiculo) REFERENCES Vehiculos(id_vehiculo),
    FOREIGN KEY (id_conductor) REFERENCES conductor(id_conductor),
    FOREIGN KEY (Id_Tipo_Servicio) REFERENCES Tipo_Servicio(Id_Tipo_Servicio),
    FOREIGN KEY (Id_Fragilidad) REFERENCES Nivel_Fragilidad(Id_Fragilidad),
    FOREIGN KEY (Id_Ubicacion) REFERENCES Ubicaciones(Id_Ubicacion)
) ENGINE=InnoDB COMMENT='Registro de servicios de envío solicitados';

-- Índices para optimización
CREATE INDEX ix_servicios_cliente ON Servicios(Id_Cliente);
CREATE INDEX ix_servicios_fecha ON Servicios(Fecha_Pedido, Fecha_Entrega);
CREATE INDEX ix_servicios_conductor ON Servicios(id_conductor);

-- ============================================
-- MÓDULO: CONTROL DE EVIDENCIAS Y DOCUMENTACIÓN
-- Fotografías, documentos y seguimiento de envíos
-- ============================================

-- Tabla: EVIDENCIA
-- Almacena fotografías y documentos del proceso de entrega
CREATE TABLE EVIDENCIA (
    id_evidencia INT PRIMARY KEY AUTO_INCREMENT,
    id_servicio INT NOT NULL,
    
    -- Tipo de evidencia capturada (foto al salir, foto al entregar o documento firmado)
    tipo_evidencia ENUM('foto_salida', 'foto_entrega', 'documento_firmado') NOT NULL,
    
    -- Nombre del archivo guardado en el sistema (ej: "foto_salida_20250124.jpg")
    nombre_archivo VARCHAR(255) NOT NULL,
    
    -- Indica si el documento/foto es legible (para verificar falsificación de documentos)
    es_legible BOOLEAN DEFAULT TRUE,
    
    -- Fecha y hora en que se capturó la evidencia (comprobante temporal)
    fecha_captura DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    -- Llave foránea hacia Servicios
    FOREIGN KEY (id_servicio) REFERENCES Servicios(Id_Servicio)
) ENGINE=InnoDB COMMENT='Evidencias fotográficas y documentos de envíos';

-- Índices para optimización
CREATE INDEX ix_evidencia_servicio ON EVIDENCIA(id_servicio);
CREATE INDEX ix_evidencia_tipo ON EVIDENCIA(tipo_evidencia);

-- Tabla: SEGUIMIENTO_CONTROL
-- Estados, calidad, incidentes y notificaciones del envío
CREATE TABLE SEGUIMIENTO_CONTROL (
    id_seguimiento INT PRIMARY KEY AUTO_INCREMENT,
    id_servicio INT NOT NULL,
    
    -- Estado actual del envío para informar al cliente (Siendo cargado, En ruta, etc.)
    estado_actual ENUM('cargando', 'en_ruta', 'en_espera', 'entregado') NOT NULL,
    
    -- Resultado de la inspección de calidad del producto
    control_calidad ENUM('aprobado', 'rechazado', 'pendiente') DEFAULT 'pendiente',
    
    -- Descripción de incidente si ocurrió (cambio de ruta, clima, mecánico, etc.)
    incidente VARCHAR(200) NULL,
    
    -- Indica si se envió notificación al cliente (correo o SMS)
    notificacion_enviada BOOLEAN DEFAULT FALSE,
    
    -- Nombre de la persona que recibió el pedido (para el reporte final)
    nombre_receptor VARCHAR(100) NULL,
    
    -- Fecha y hora del registro (para historial y reporte de entrega)
    fecha_hora DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    -- Llave foránea hacia Servicios
    FOREIGN KEY (id_servicio) REFERENCES Servicios(Id_Servicio)
) ENGINE=InnoDB COMMENT='Seguimiento y control de estados de envíos';

-- Índices para optimización
CREATE INDEX ix_seguimiento_servicio ON SEGUIMIENTO_CONTROL(id_servicio);
CREATE INDEX ix_seguimiento_estado ON SEGUIMIENTO_CONTROL(estado_actual);
CREATE INDEX ix_seguimiento_fecha ON SEGUIMIENTO_CONTROL(fecha_hora);
